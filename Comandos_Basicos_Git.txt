
# Comandos básicos de Git para usar dentro de Visual Studio Code desde terminal:

# 1. Configuración inicial
git config --global user.name "Tu Nombre"
git config --global user.email "tuemail@example.com"

# 2. Clonar un repositorio
git clone https://github.com/usuario/nombre-repositorio.git

# 3. Inicializar un nuevo repositorio
git init

# 4. Ver el estado actual del repositorio
git status

# 5. Añadir archivos al área de preparación
git add nombre-archivo
git add .  # Añadir todos los archivos modificados

# 6. Realizar un commit
git commit -m "Mensaje descriptivo del cambio"

# 7. Subir cambios al repositorio remoto
git push origin main

# 8. Obtener cambios del repositorio remoto
git pull origin main

# 9. Crear una nueva rama
git checkout -b nombre-de-la-rama

# 10. Cambiar a una rama existente
git checkout nombre-de-la-rama

# 11. Subir una nueva rama al repositorio remoto
git push -u origin nombre-de-la-rama

# 12. Fusionar ramas (merge)
git merge nombre-de-la-rama

# 13. Ver el historial de commits
git log

# 14. Cancelar cambios antes del commit
git checkout -- nombre-archivo  # Revertir cambios en un archivo
git reset HEAD nombre-archivo   # Sacar un archivo del área de preparación

# 15. Configurar claves SSH (opcional)
ssh-keygen -t ed25519 -C "tuemail@example.com"
